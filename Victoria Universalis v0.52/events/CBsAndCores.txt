# Gain Core event
province_event = {
	id = 2560
	title = "EVTNAME2560"
	desc = "EVTDESC2560"
	
	trigger = {
		NOT = { is_core = THIS }
		controlled_by = THIS
		is_colonial = no
		OR = {
			is_primary_culture = yes
			is_accepted_culture = yes
			any_neighbor_province = {
				AND = {
					is_core = THIS
					controlled_by = THIS
				}
			}
		}
		owner = { tag = THIS }
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.4
			OR = {
				is_primary_culture = yes
				is_accepted_culture = yes
			}
		}
		modifier = {
			factor = 2
			average_militancy = 3
		}
		modifier = {
			factor = 2
			average_militancy = 5
		}
		modifier = {
			factor = 10
			average_militancy = 9
		}
	}
	
	option = {
		name = "EVTOPTA2560"
		add_core = THIS
	}
}
