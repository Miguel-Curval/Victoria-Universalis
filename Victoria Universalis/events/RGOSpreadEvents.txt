# Tractors
province_event = {
	id = 29900
	title = "EVTNAME29900"
	desc = "EVTDESC29900"
	
	trigger = {
		is_colonial = no
		OR = {
			trade_goods = cotton
			trade_goods = dye
			trade_goods = wool
			trade_goods = silk
			trade_goods = cattle
			trade_goods = fish
			trade_goods = fruit
			trade_goods = grain
			trade_goods = tobacco
			trade_goods = tea
			trade_goods = coffee
			trade_goods = opium
		}	
		NOT = {
			has_province_modifier = local_tractors
		}
		owner = {
			invention = tractors
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "EVTOPTA29900"
		owner = {
			any_owned = {
				limit = {
					NOT = { has_province_modifier = local_tractors }					
					is_colonial = no 
					OR = {
						trade_goods = cotton
						trade_goods = dye
						trade_goods = wool
						trade_goods = silk
						trade_goods = cattle
						trade_goods = fish
						trade_goods = fruit
						trade_goods = grain
						trade_goods = tobacco
						trade_goods = tea
						trade_goods = coffee
						trade_goods = opium
					}	
				}						
				add_province_modifier = {
					name = local_tractors
					duration = -1
				}
			}
		}
	}
}
